40  public void Add(IEvaluator ev) {

52  bool DoesNotOverlapWithAnyEvaluator(IEvaluator ev) {
53  foreach(var cev in Evaluators)
54  if(ev.XInterval.Overlaps(cev.XInterval))
55  if(ev.XInterval.Adjacent(cev.XInterval))
56  continue;
57  else
58  return false;
59 
60  return true;
61  }

63  void UpdateXyPoints() {

71  void SingleEvaluatorXyPointsUpdate() {
72  Xa = Evaluators[0].PtA.X;
73  Ya = Evaluators[0].PtA.Y;
74  Xb = Evaluators[0].PtB.X;

78  void MultiEvaluatorXyPointsUpdate() {
79  foreach(var ev in Evaluators) {
80  if(Xa >= ev.MinX) {
81  Xa = ev.MinX;
82  Ya = ev.PtA.Y;
83  }
84  if(Xb <= ev.MaxX) {
85  Xb = ev.MaxX;
86  Yb = ev.PtB.Y;
87  }
88  }

98  IEvaluator FindEvaluator(float x) {
99  int evCount = Evaluators.Count;
100  if(x.InInterval(XInterval))
101  return FindInternalEvaluator(x);
102 
103  if(x.AboveInterval(XInterval))
104  return Evaluators[evCount - 1];
105 
106  return x.BelowInterval(XInterval) ? Evaluators[0] : null;
107  }
108 
109  IEvaluator FindInternalEvaluator(float x) {
110  int evCount = Evaluators.Count;
111  for(int i = 0; i < evCount; i++)
112  if(x.InInterval(Evaluators[i].XInterval))
113  return Evaluators[i];
114  // x is in a "hole"
115  return null;
116  }
117 
118  float LinearHoleInterpolator(float x) {
119  var lrev = FindLeftAndRightInterpolators(x);
120  var xl = lrev.Key.MaxX;
121  var yl = lrev.Key.Evaluate(xl);
122  var xr = lrev.Value.MinX;
123  var yr = lrev.Value.Evaluate(xr);
124  var alpha = (x - xl) / (xr - xl);
125  return yl + alpha * (yr - yl);
126  }
127 
128  KeyValuePair<IEvaluator, IEvaluator> FindLeftAndRightInterpolators(float x) {
129  int evCount = Evaluators.Count;
130  IEvaluator lev = null;
131  IEvaluator rev = null;
132  for(int i = 0; i < evCount - 1; i++) {
133  lev = Evaluators[i];
134  rev = Evaluators[i + 1];
135  if(x.AboveInterval(lev.XInterval) &&
136  x.BelowInterval(rev.XInterval))
137  break;
138  }
139 
140  return new KeyValuePair<IEvaluator, IEvaluator>(lev, rev);
141  }

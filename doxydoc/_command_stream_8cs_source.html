92 
93  do {
94  if(CanGetNextCommand()) {
95  CalculateOverdueUpdates();

214  public void Pause() {
215  if(_isActive == false)
216  return;
217 
218  float time = CrTime.Time;
219  float num = NextExecution - time;
220  LastExecution = num;
221  NextExecution = float.PositiveInfinity;
222  Queue.UpdatePriority(this, NextExecution);
223  }
224 
225  public void Resume() {
226  if(_isActive == false)
227  return;
228 
229  float time = CrTime.Time;
230  LastExecution = time;
231  NextExecution = time + Command.ExecutionDelay;
232  Queue.UpdatePriority(this, NextExecution);
233  }
234 
235  void AddSelfToQueue() {
236  float time = CrTime.Time;
237  LastExecution = time;
238  NextExecution = time + Command.ExecutionDelay;
239  Queue.Add(this, NextExecution);
240  }
241 
242  void RemoveSelfFromQueue() {
243  if(Queue.Peek() == this)
244  Queue.Dequeue();
245  else
246  Queue.Remove(this);
247  }

56  public bool Contains(T point) {
57  if(LowerBound.GetType() != typeof(T)
58  ||
59  UpperBound.GetType() != typeof(T))
60  throw new ArgumentException("Type mismatch", "point");
61 
62  var lower = LowerBoundType == IntervalType.Open
63  ? LowerBound.CompareTo(point) < 0
64  : LowerBound.CompareTo(point) <= 0;
65  var upper = UpperBoundType == IntervalType.Open
66  ? UpperBound.CompareTo(point) > 0
67  : UpperBound.CompareTo(point) >= 0;
68 
69  return lower && upper;
70  }

87  public Interval(T lowerbound, T upperbound,
88  IntervalType lowerboundIntervalType = IntervalType.Closed,
89  IntervalType upperboundIntervalType = IntervalType.Closed)
90  : this() {
91  var a = lowerbound;
92  var b = upperbound;
93  var comparison = a.CompareTo(b);
94 
95  if(comparison > 0) {
96  a = upperbound;
97  b = lowerbound;
98  }
99 
100  LowerBound = a;
101  UpperBound = b;
102  LowerBoundType = lowerboundIntervalType;
103  UpperBoundType = upperboundIntervalType;
104  }

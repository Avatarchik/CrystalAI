77  public bool IsBinaryHeapValid() {
78  for(int i = 1; i < _nodes.Length; i++)
79  if(_nodes[i] != null) {
80  int childLeftIndex = 2 * i;

82  _nodes[childLeftIndex] != null &&

232  QueueNode Remove(QueueNode node) {
233  if(node == null)
234  return null;
235 
236  QueueNode rNode;
237  // Check if the node is the last since we can remove this one quickly.
238  if(node.QueueIndex == _numNodes) {
239  rNode = _nodes[_numNodes];
240  _nodes[_numNodes] = null;
241  _numNodes--;
242  return rNode;
243  }
244 
245  QueueNode oldLastNode = _nodes[_numNodes];
246  Swap(node, oldLastNode);
247  rNode = _nodes[_numNodes];
248  _nodes[_numNodes] = null;
249  _numNodes--;
250 
251  // Reposition the previous last node up or down as appropriate.
252  OnNodeUpdated(oldLastNode);
253  return rNode;
254  }

276  void CascadeUp(QueueNode node) {
277  int parent = node.QueueIndex / 2;
278  while(parent >= 1) {
279  QueueNode parentNode = _nodes[parent];
280  if(HasHigherPriority(parentNode, node))
281  break;
282 
283  // Node has lower priority value, so move it up the heap
284  Swap(node, parentNode);
285  parent = node.QueueIndex / 2;
286  }
287  }

293  void CascadeDown(QueueNode node) {
294  QueueNode newParent;
295  int finalQueueIndex = node.QueueIndex;
296  while(true) {
297  newParent = node;
298  int childLeftIndex = 2 * finalQueueIndex;
299 
300  //Check if the left-child is higher-priority than the current node
301  if(childLeftIndex > _numNodes) {
302  //This could be placed outside the loop, but then we'd have to check newParent != node twice
303  node.QueueIndex = finalQueueIndex;
304  _nodes[finalQueueIndex] = node;
305  break;
306  }
307 
308  QueueNode childLeft = _nodes[childLeftIndex];
309  if(HasHigherPriority(childLeft, newParent))
310  newParent = childLeft;
311 
312  //Check if the right-child is higher-priority than either the current node or the left child
313  int childRightIndex = childLeftIndex + 1;
314  if(childRightIndex <= _numNodes) {
315  QueueNode childRight = _nodes[childRightIndex];
316  if(HasHigherPriority(childRight, newParent))
317  newParent = childRight;
318  }
319 
320  //If either of the children has higher (smaller) priority, swap and continue cascading
321  if(newParent != node) {
322  //Move new parent to its new index. node will be moved once, at the end
323  //Doing it this way is one less assignment operation than calling Swap()
324  _nodes[finalQueueIndex] = newParent;
325 
326  int temp = newParent.QueueIndex;
327  newParent.QueueIndex = finalQueueIndex;
328  finalQueueIndex = temp;
329  } else {
330  //See note above
331  node.QueueIndex = finalQueueIndex;
332  _nodes[finalQueueIndex] = node;
333  break;
334  }
335  }
336  }

358  void OnNodeUpdated(QueueNode node) {
359  //Bubble the updated node up or down as appropriate
360  int parentIndex = node.QueueIndex / 2;
361  QueueNode parentNode = _nodes[parentIndex];
362 
363  if(parentIndex > 0 &&
364  HasHigherPriority(node, parentNode))
365  CascadeUp(node);
366  else
367  CascadeDown(node);
368  }
